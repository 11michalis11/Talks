\begin{frame}
    \frametitle{Waiting Times}

    \[        
    W = \frac{\sum_{(u,v) \in S_A} w(u,v) \pi_{(u,v)}}{\sum_{(u,v) \in S_A} \pi_{(u,v)}}
    \]
    \vspace{1cm}
    \[
    W = \frac{\lambda_o P(L'_o)}{\lambda_a P(L'_a) + \lambda_o P(L'_o)} W^{(o)} + \frac{\lambda_a P(L'_a)}{\lambda_a P(L'_a) + \lambda_o P(L'_o)} W^{(a)}
    \]

\end{frame}
